  \hyphenpenalty=10000%
  \exhyphenpenalty=10000%
}

  \mathchardef\template@hyphenation@internal@hyphenpenalty=\hyphenpenalty%
  \mathchardef\template@hyphenation@internal@exhyphenpenalty=\exhyphenpenalty%
}

  \hyphenpenalty=\template@hyphenation@internal@hyphenpenalty%
  \exhyphenpenalty=\template@hyphenation@internal@exhyphenpenalty%
}
\makeatother

% It sets Russian and English alphabets for letter-based counters.
% It likely needs package babel.
% The letters Ё, З, Й, О, Ч, Ъ, Ы, Ь are skipped.
% See "ГОСТ Р 2.105-2019" p. 6.3.5.
\makeatletter
\def\russian@alph#1{\ifcase#1\or а\or б\or в\or г\or д\or е\or ж\or и\or к\or л\or м\or н\or п\or р\or с\or т\or у\or ф\or х\or ц\or ш\or щ\or э\or ю\or я\else\@ctrerr\fi}
\def\russian@Alph#1{\ifcase#1\or А\or Б\or В\or Г\or Д\or Е\or Ж\or И\or К\or Л\or М\or Н\or П\or Р\or С\or Т\or У\or Ф\or Х\or Ц\or Ш\or Щ\or Э\or Ю\or Я\else\@ctrerr\fi}
\makeatother

% It sets Russian and English alphabets for enumerations.
% It likely needs package babel.
% The letters I, O are skipped.
% See "ГОСТ Р 2.105-2019" p. 6.3.5.
\makeatletter
\def\@alph#1{\ifcase#1\or a\or b\or c\or d\or e\or f\or g\or h\or j\or k\or l\or m\or n\or p\or q\or r\or s\or t\or u\or v\or w\or x\or y\or z\else\@ctrerr\fi}
\def\@Alph#1{\ifcase#1\or A\or B\or C\or D\or E\or F\or G\or H\or J\or K\or L\or M\or N\or P\or Q\or R\or S\or T\or U\or V\or W\or X\or Y\or Z\else\@ctrerr\fi}
\makeatother

% Set default image width and height.
% If image fits, it has original size.
% If image doesn't fit, it is constrained by \template@image@width and \template@image@height.
% Extremely narrow images overflow, you should set their height manually.
\makeatletter
\def\template@image@width{0.75\linewidth}   % 75% width of the line in the current environment
\def\template@image@height{1.00\textheight} % 100% height of the text area in the page
\def\template@image@internal@width{\ifdim\Gin@nat@width>\template@image@width \template@image@width\else\Gin@nat@width\fi}
\def\template@image@internal@height{\ifdim\Gin@nat@height>\template@image@height \template@image@height\else\Gin@nat@height\fi}
\setkeys{Gin}{width=\template@image@internal@width,height=\template@image@internal@height,keepaspectratio}
\makeatother
